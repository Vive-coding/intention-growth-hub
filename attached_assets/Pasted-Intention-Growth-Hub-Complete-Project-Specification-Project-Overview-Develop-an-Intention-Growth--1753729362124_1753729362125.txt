Intention Growth Hub - Complete Project Specification
Project Overview
Develop an "Intention Growth Hub" web application featuring a sophisticated dashboard with 6 life areas tracking (Mental Health, Physical Health, Social, Productivity, Nutrition, Investments), goals management, AI insights, and community features. The application should match the advanced design created in Lovable with proper styling, charts, and comprehensive user interface components.

Technology Stack
Frontend: React 18.3.1 with TypeScript
Build Tool: Vite for fast development and optimized builds
Styling: Tailwind CSS with shadcn/ui component library
State Management: React hooks for local state, TanStack React Query for server state
Routing: Wouter for client-side navigation
UI Components: Radix UI primitives with custom styling
Backend: Express.js with TypeScript
Database: PostgreSQL with UUID-based primary keys
ORM: Drizzle ORM for database interactions
AI Integration: OpenAI GPT-4 for insights generation
Charts: Recharts for data visualization
Icons: Lucide React
Database Schema (PostgreSQL with UUIDs)
Core User Management
-- Users table
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    gpt_oauth_sub VARCHAR(255), -- For GPT integration
    api_token TEXT, -- For external API access
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
Life Metrics System
-- Life metric definitions (customizable per user)
CREATE TABLE life_metric_definitions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(255) NOT NULL, -- Mental Health, Physical Health, etc.
    description TEXT,
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
Goals & Progress Tracking
-- Goal definitions (templates)
CREATE TABLE goal_definitions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    category VARCHAR(100), -- Links to life metric areas
    unit VARCHAR(50), -- minutes, hours, count, etc.
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
-- Goal instances (active goals with targets)
CREATE TABLE goal_instances (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    goal_definition_id UUID REFERENCES goal_definitions(id) ON DELETE CASCADE,
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    target_value INTEGER NOT NULL,
    start_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    target_date TIMESTAMP,
    status VARCHAR(20) DEFAULT 'active', -- active, completed, paused, cancelled
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
-- Progress tracking
CREATE TABLE goal_progress_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    goal_instance_id UUID REFERENCES goal_instances(id) ON DELETE CASCADE,
    progress_value INTEGER NOT NULL,
    notes TEXT,
    recorded_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    journal_entry_id UUID REFERENCES journal_entries(id) -- Optional link to journal
);
Habits System
-- Habit definitions
CREATE TABLE habit_definitions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    habit_type VARCHAR(50), -- daily, weekly, monthly
    frequency_type VARCHAR(20), -- times_per_day, times_per_week, etc.
    target_frequency INTEGER, -- how many times per period
    linked_goal_instance_ids JSONB, -- Array of linked goal IDs
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
-- Habit completion tracking
CREATE TABLE habit_completion_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    habit_id UUID REFERENCES habit_definitions(id) ON DELETE CASCADE,
    completion_datetime TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed BOOLEAN DEFAULT true,
    notes TEXT,
    journal_entry_id UUID REFERENCES journal_entries(id) -- Optional link to journal
);
Journaling System
-- Journal entries
CREATE TABLE journal_entries (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    content TEXT NOT NULL,
    mood_rating INTEGER CHECK (mood_rating >= 1 AND mood_rating <= 10),
    tags JSONB, -- Array of tags for categorization
    source VARCHAR(50) DEFAULT 'manual', -- manual, voice, gpt_chat, import
    gpt_conversation_id VARCHAR(255), -- Link to GPT conversation
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
AI Insights System
-- AI-generated insights
CREATE TABLE insights (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    content TEXT NOT NULL,
    insight_type VARCHAR(50), -- pattern, recommendation, achievement, warning, trend
    confidence_score INTEGER CHECK (confidence_score >= 0 AND confidence_score <= 100),
    data_source_ids JSONB, -- References to journal entries, goals, habits that generated this
    is_active BOOLEAN DEFAULT true,
    relevance_score INTEGER CHECK (relevance_score >= 0 AND relevance_score <= 100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
